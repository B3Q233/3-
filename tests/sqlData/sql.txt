create database AIChat;
use AIChat;
create table users
(
    id       int auto_increment
        primary key,
    email    varchar(30)  not null,
    api_key  varchar(255) not null,
    password varchar(20)  not null,
    username varchar(20)  not null,
    quota    int          null,
    constraint email
        unique (email),
    constraint ix_unique_users_id
        unique (id),
    constraint username
        unique (username)
);
